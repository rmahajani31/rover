# micro-ROS agent on the Pi
ros2 run micro_ros_agent micro_ros_agent serial --dev /dev/tty_rover_esp | tee micro_ros.txt

# Create launch output directory
mkdir -p out

# Livox driver on the Jetson for baseline scan projection modes
ros2 launch livox_ros_driver2 rviz_MID360_launch.py

# Livox driver on the Jetson for FAST-LIO2 and custom lidar odometry modes
ros2 launch livox_ros_driver2 msg_MID360_launch.py

# Build Jetson-side lidar packages
colcon build --packages-select fast_lio custom_fastlio_preprocess custom_livox_costmap_projection custom_icp_odom custom_imu_propagator custom_scan_to_map_odom custom_lidar_deskew custom_ikd_tree_backend custom_lio_ekf fastlio2_nav2_adapter livox_cloud_to_scan bringup

# Build Pi-side packages while skipping Jetson-only packages whose dependencies are not installed on the Pi
colcon build --packages-skip fast_lio fastlio2_nav2_adapter custom_fastlio_preprocess custom_livox_costmap_projection custom_icp_odom custom_imu_propagator custom_scan_to_map_odom custom_lidar_deskew custom_ikd_tree_backend custom_lio_ekf

# Baseline Jetson scan projection
ros2 launch bringup jetson.launch.py | tee out/jetson.txt

# Pi mapping with gamepad control and goBILDA Pinpoint v2 rover odometry
ros2 launch bringup mapping.launch.py | tee out/mapping.txt

# Save the current slam_toolbox map
ros2 service call /slam_toolbox/save_map slam_toolbox/srv/SaveMap "{name: {data: '/home/rmahajani/Documents/projects/rover/ros2_ws/maps/rover_map'}}"

# Pi navigation with goBILDA Pinpoint v2 rover odometry
ros2 launch bringup pi_nav2_livox.launch.py map:=/home/rmahajani/Documents/projects/rover/ros2_ws/maps/rover_map.yaml | tee out/pi_nav2_livox.txt

# FAST-LIO2 shadow mode on the Jetson
ros2 launch bringup fastlio_shadow.launch.py | tee out/fastlio_shadow.txt

# FAST-LIO2 Nav2 odometry on the Jetson
ros2 launch bringup fast_lio2_nav2.launch.py | tee out/fast_lio2_nav2.txt

# FAST-LIO2 Nav2 odometry with custom Livox preprocessing on the Jetson
ros2 launch bringup pointcloud_preprocess_fast_lio2_nav2.launch.py | tee out/pointcloud_preprocess_fast_lio2_nav2.txt

# Custom scan-to-scan ICP shadow mode on the Jetson
ros2 launch bringup scan_to_scan_icp.launch.py | tee out/scan_to_scan_icp.txt

# Custom scan-to-map primary odometry on the Jetson
ros2 launch bringup scan_to_map_primary_odom.launch.py | tee out/scan_to_map_primary_odom.txt

# FAST-LIO2 odometry with upgraded Livox obstacle cloud on the Jetson
ros2 launch bringup costmap_projection.launch.py | tee out/costmap_projection.txt

# IMU-assisted scan-to-map odometry with upgraded Livox obstacle cloud on the Jetson
ros2 launch bringup scan_to_map_costmap_projection.launch.py | tee out/scan_to_map_costmap_projection.txt

# Deskewed scan-to-map odometry with upgraded Livox obstacle cloud on the Jetson
ros2 launch bringup scan_to_map_deskew_costmap_projection.launch.py | tee out/scan_to_map_deskew_costmap_projection.txt

# Error-state EKF odometry with upgraded Livox obstacle cloud on the Jetson
ros2 launch bringup lio_ekf_deskew_costmap_projection.launch.py | tee out/lio_ekf_deskew_costmap_projection.txt

# Pi Nav2 consuming Jetson FAST-LIO2 or scan-to-map /nav2_odom with /scan_from_livox costmaps
ros2 launch bringup pi_fast_lio2_nav2_main.launch.py map:=/home/rmahajani/Documents/projects/rover/ros2_ws/maps/rover_map.yaml | tee out/pi_fast_lio2_nav2_main.txt

# Pi Nav2 consuming Jetson /nav2_odom, /scan_from_livox for AMCL, and /custom/obstacle_cloud for local obstacles
ros2 launch bringup pi_fast_lio2_costmap_projection_nav2.launch.py map:=/home/rmahajani/Documents/projects/rover/ros2_ws/maps/rover_map.yaml | tee out/pi_fast_lio2_costmap_projection_nav2.txt
